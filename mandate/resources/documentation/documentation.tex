\section{Assets}

Assets to use

\begin{lstlisting}[style=sql]
	CREATE TABLE IF NOT EXISTS block (
		id INT PRIMARY KEY,
		difficulty INT,
		tx_hash BINARY(32),
		nonce BINARY(32),
		miner BINARY(32),
		mined DATETIME
	);
\end{lstlisting}

\begin{lstlisting}[style=sql]
	CREATE TABLE IF NOT EXISTS tx (
		block_id INT,
		sender_pub BINARY(32),
		recipient BINARY(32),
		amount INT,
		timestamp DATETIME,
		last_tx_hash BINARY(32),
		signature BINARY(64) 
	);
\end{lstlisting}

\begin{lstlisting}[style=sql]
	CREATE TABLE IF NOT EXISTS wallet (
		address BINARY(32),
		amount INT
	)
\end{lstlisting}

\begin{lstlisting}[style=sql]
	CREATE TABLE IF NOT EXISTS keyCache (
		pub_key BINARY(32),
		address BINARY(32)
	)
\end{lstlisting}
